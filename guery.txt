FROM employees e

GROUP BY d.name;

FROM
( SELECT e.salary as salary,d.branch_id 
 FROM employees e 
 JOIN departments d 
 ON e.department_id=d.id) temp 
 JOIN branches b 
 ON temp.branch_id =b.id
 GROUP BY b.name
 HAVING(b.name='Isfahan');
